-- â”‚ Query 6: Analyze Decision Performance by Rule Citation                    â”‚
-- â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”˜
SELECT 
    dl.CitedRuleIds,
    COUNT(*) AS TotalDecisions,
    SUM(CASE WHEN dl.Action = 'BUY' THEN 1 ELSE 0 END) AS BuyCount,
    SUM(CASE WHEN dl.Action = 'SELL' THEN 1 ELSE 0 END) AS SellCount,
    SUM(CASE WHEN dl.Action = 'HOLD' THEN 1 ELSE 0 END) AS HoldCount,
    AVG(dl.PortfolioValueAfter - dl.PortfolioValueBefore) AS AvgPortfolioChange
FROM DecisionLogs dl
WHERE dl.Timestamp >= DATEADD(DAY, -30, GETUTCDATE())
GROUP BY dl.CitedRuleIds
ORDER BY TotalDecisions DESC;


-- â”Œâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”
-- â”‚ Query 7: Get Current Active Regime for Each Asset                         â”‚
-- â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”˜
WITH LatestRegimes AS (
    SELECT 
        Asset,
        RegimeId,
        DetectedAt,
        Volatility,
        ROW_NUMBER() OVER (PARTITION BY Asset ORDER BY DetectedAt DESC) AS rn
    FROM DetectedRegimes
)
SELECT 
    Asset,
    RegimeId,
    DetectedAt,
    Volatility
FROM LatestRegimes
WHERE rn = 1 AND IsActive = 1;


-- â”Œâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”
-- â”‚ Query 8: Audit Trail - Full Decision Context                              â”‚
-- â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”˜
SELECT 
    dl.Id AS DecisionId,
    a.Name AS AgentName,
    dl.Timestamp,
    dl.Action,
    dl.Asset,
    dl.Quantity,
    dl.Rationale,
    dl.CitedRuleIds,
    dl.DetectedRegime,
    dl.SubgraphSnapshot,
    dl.MarketConditions,
    dl.PortfolioValueBefore,
    dl.PortfolioValueAfter,
    (dl.PortfolioValueAfter - dl.PortfolioValueBefore) AS PnL
FROM DecisionLogs dl
INNER JOIN Agents a ON dl.AgentId = a.Id
WHERE dl.Id = @DecisionId;
